# Configuration Google OAuth pour Laravel
# Ajoutez ces variables à votre fichier .env

# Google OAuth Configuration
GOOGLE_CLIENT_ID=your-google-client-id
GOOGLE_CLIENT_SECRET=your-google-client-secret
GOOGLE_REDIRECT_URI=http://localhost:8000/api/auth/google/callback

# Instructions pour obtenir les credentials Google OAuth:
# 1. Allez sur https://console.developers.google.com/
# 2. Créez un nouveau projet ou sélectionnez un projet existant
# 3. Activez l'API Google+ 
# 4. Allez dans "Credentials" et créez un "OAuth 2.0 Client ID"
# 5. Configurez les URIs de redirection autorisés:
#    - http://localhost:8000/api/auth/google/callback (pour le développement)
#    - https://votre-domaine.com/api/auth/google/callback (pour la production)
# 6. Copiez le Client ID et Client Secret dans votre .env
